### 1. CryptoKit端到端加密核心

```swift
import CryptoKit
import Foundation

/// VaultNote加密引擎 — 所有数据加密/解密的唯一入口
final class VaultCrypto {
    
    // MARK: - 密钥派生（从用户密码生成256位密钥）
    
    /// 使用PBKDF2从密码派生密钥
    /// - Parameters:
    ///   - password: 用户主密码
    ///   - salt: 随机盐值（存储在Keychain中）
    /// - Returns: 256位对称密钥
    static func deriveKey(from password: String, salt: Data) -> SymmetricKey {
        let derivedKey = HKDF<SHA256>.deriveKey(
            inputKeyMaterial: SymmetricKey(data: Data(password.utf8)),
            salt: salt,
            info: Data("VaultNote-E2E-Key".utf8),
            outputByteCount: 32
        )
        return derivedKey
    }
    
    // MARK: - AES-GCM加密（每条笔记独立加密）
    
    /// 加密笔记内容
    /// - Parameters:
    ///   - plaintext: 原始Markdown文本
    ///   - key: 对称密钥
    /// - Returns: 加密数据（nonce + ciphertext + tag）
    static func encrypt(_ plaintext: String, with key: SymmetricKey) throws -> Data {
        let sealedBox = try AES.GCM.seal(
            Data(plaintext.utf8),
            using: key
        )
        return sealedBox.combined! // nonce(12B) + ciphertext + tag(16B)
    }
    
    /// 解密笔记内容
    /// - Parameters:
    ///   - encryptedData: 加密数据
    ///   - key: 对称密钥
    /// - Returns: 原始Markdown文本
    static func decrypt(_ encryptedData: Data, with key: SymmetricKey) throws -> String {
        let sealedBox = try AES.GCM.SealedBox(combined: encryptedData)
        let decryptedData = try AES.GCM.open(sealedBox, using: key)
        return String(data: decryptedData, encoding: .utf8)!
    }
    
    // MARK: - 密钥安全存储（Keychain）
    
    /// 将加密盐值存储到Keychain
    static func storeSalt(_ salt: Data, key: String = "com.vaultnote.salt") {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecValueData as String: salt
        ]
        SecItemDelete(query as CFDictionary)
        SecItemAdd(query as CFDictionary, nil)
    }
    
    /// 从Keychain读取盐值
    static func retrieveSalt(key: String = "com.vaultnote.salt") -> Data? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecReturnData as String: true
        ]
        var result: AnyObject?
        SecItemCopyMatching(query as CFDictionary, &result)
        return result as? Data
    }
}
```

### 2. CoreData数据模型（加密存储层）

```swift
import CoreData

/// VaultNote Core Data模型定义
/// 注意：noteBody字段存储的是加密后的Data，而非明文

// MARK: - 实体定义（VNNote.xcdatamodeld）
/*
 Entity: VNNote
 ┌──────────────┬─────────────┬─────────────────────┐
 │ Attribute    │ Type        │ 说明                 │
 ├──────────────┼─────────────┼─────────────────────┤
 │ id           │ UUID        │ 唯一标识             │
 │ title        │ String      │ 笔记标题（明文，用于搜索索引）│
 │ encryptedBody│ Binary      │ AES-GCM加密后的Markdown │
 │ tags         │ Transformable│ [String] 标签数组    │
 │ isPinned     │ Boolean     │ 是否置顶             │
 │ isFavorite   │ Boolean     │ 是否收藏             │
 │ createdAt    │ Date        │ 创建时间             │
 │ updatedAt    │ Date        │ 更新时间             │
 │ folder       │ String      │ 所属文件夹           │
 │ nonce        │ Binary      │ 加密nonce（12字节）   │
 └──────────────┴─────────────┴─────────────────────┘
 */

/// CoreData持久化控制器
final class VaultDataController: ObservableObject {
    
    static let shared = VaultDataController()
    
    let container: NSPersistentCloudKitContainer
    
    init() {
        container = NSPersistentCloudKitContainer(name: "VaultNote")
        
        // 关键：仅在用户开启iCloud同步时启用CloudKit
        // 默认仅本地存储
        let storeURL = URL.storeDirectory()
            .appendingPathComponent("VaultNote.sqlite")
        
        let localDescription = NSPersistentStoreDescription(url: storeURL)
        localDescription.configuration = "Local" // 本地配置
        
        container.persistentStoreDescriptions = [localDescription]
        
        container.loadPersistentStores { _, error in
            if let error = error {
                fatalError("CoreData failed: \(error)")
            }
        }
        container.viewContext.automaticallyMergesChangesFromParent = true
    }
    
    // MARK: - 启用iCloud E2E同步
    
    /// 用户主动开启iCloud同步时调用
    func enableCloudSync() {
        guard !UserDefaults.standard.bool(forKey: "cloudSyncEnabled") else { return }
        
        let cloudDescription = NSPersistentCloudKitContainer
            .StoreDescription(.init(url: URL.storeDirectory()
                .appendingPathComponent("VaultNote-Cloud.sqlite")))
        cloudDescription.configuration = "Cloud"
        cloudDescription.cloudKitContainerOptions = 
            NSPersistentCloudKitContainer.Options(containerIdentifier: "iCloud.com.vaultnote")
        
        // NSPersistentCloudKitContainer自动处理E2E加密
        // 因为我们的noteBody已经是加密Data，CloudKit仅传输密文
        
        container.persistentStoreDescriptions.append(cloudDescription)
        try? container.persistentStoreCoordinator.addPersistentStore(
            with: cloudDescription
        )
        UserDefaults.standard.set(true, forKey: "cloudSyncEnabled")
    }
}
```

### 3. Face ID/Touch ID认证

```swift
import LocalAuthentication

/// 生物识别认证管理器
final class VaultAuthManager: ObservableObject {
    
    @Published var isUnlocked = false
    
    /// 验证用户身份
    /// - Parameter reason: 验证原因描述
    func authenticate(reason: String = "Unlock VaultNote") async -> Bool {
        let context = LAContext()
        var error: NSError?
        
        // 检查设备是否支持生物识别
        guard context.canEvaluatePolicy(.deviceOwnerAuthenticationWithBiometrics, error: &error) else {
            // 降级到设备密码
            return await authenticateWithPasscode()
        }
        
        do {
            let success = try await context.evaluatePolicy(
                .deviceOwnerAuthenticationWithBiometrics,
                localizedReason: reason
            )
            await MainActor.run { self.isUnlocked = success }
            return success
        } catch {
            // Face ID失败，降级到设备密码
            return await authenticateWithPasscode()
        }
    }
    
    /// 降级：使用设备密码认证
    private func authenticateWithPasscode() async -> Bool {
        let context = LAContext()
        do {
            let success = try await context.evaluatePolicy(
                .deviceOwnerAuthentication, // 包含密码回退
                localizedReason: "Enter your passcode to unlock VaultNote"
            )
            await MainActor.run { self.isUnlocked = success }
            return success
        } catch {
            return false
        }
    }
}
```

```swift
        } catch {
            return AttributedString(markdown)
        }
    }
```

### 5. 全文搜索实现

```swift
import CoreData

/// 全文搜索引擎（本地索引，无需网络）
final class VaultSearchEngine {
    
    /// 搜索笔记（标题+标签+解密后内容）
    /// - Parameters:
    ///   - query: 搜索关键词
    ///   - context: CoreData上下文
    ///   - cryptoKey: 解密密钥
    /// - Returns: 匹配的笔记数组
    static func search(
        query: String,
        context: NSManagedObjectContext,
        cryptoKey: SymmetricKey? = nil
    ) -> [VNNote] {
        let request: NSFetchRequest<VNNote> = VNNote.fetchRequest()
        
        // 先在标题和标签中搜索（无需解密，速度快）
        let titlePredicate = NSPredicate(
            format: "title CONTAINS[cd] %@", query
        )
        let tagsPredicate = NSPredicate(
            format: "tags CONTAINS[cd] %@", query
        )
        request.predicate = NSCompoundPredicate(
            orPredicateWithSubpredicates: [titlePredicate, tagsPredicate]
        )
        request.sortDescriptors = [
            NSSortDescriptor(keyPath: \VNNote.updatedAt, ascending: false)
        ]
        
        var results = (try? context.fetch(request)) ?? []
        
        // 如果有密钥，搜索加密内容（性能优化：仅搜索最近100条）
        if let key = cryptoKey {
            let contentRequest: NSFetchRequest<VNNote> = VNNote.fetchRequest()
            contentRequest.sortDescriptors = [
                NSSortDescriptor(keyPath: \VNNote.updatedAt, ascending: false)
            ]
            contentRequest.fetchLimit = 100
            
            let allNotes = (try? context.fetch(contentRequest)) ?? []
            for note in allNotes {
                if let encrypted = note.encryptedBody,
                   let decrypted = try? VaultCrypto.decrypt(encrypted, with: key),
                   decrypted.localizedCaseInsensitiveContains(query),
                   !results.contains(where: { $0.id == note.id }) {
                    results.append(note)
                }
            }
        }
        
        return results
    }
}
```

#### 市场定价现状

| 竞品 | 定价模式 | 价格 | 用户痛点 |

| Obsidian Sync | 订阅 | $5/月 | 免费版无同步 |
| Evernote | 订阅 | $14.99/月 | 极贵，免费版50条限制 |
| GoodNotes | 一次性 | $9.99 | 但仅限手写笔记 |
| Joplin | 免费 | $0 | 但UI差，非原生 |
| LightNotes | 免费 | $0 | 无加密，无同步 |
| Apple Notes | 免费 | $0 | 无加密，iCloud非E2E |

#### 核心定价洞察

1. **订阅疲劳是真实存在的** — Reddit用户明确说："If it was one price-to-own then that would make sense"
2. **隐私用户不愿月月付费** — 他们选择离线App就是不想被持续跟踪，订阅模式与此矛盾
3. **$4.99是甜蜜点** — 低于$5的心理门槛，比Bear($2.99×12=$35.88/年)便宜7倍
4. **免费版必须有足够价值** — 10条笔记足以让用户评估，但不构成完整使用

## 🏆 竞争优势总结

```
┌──────────────────────────────────────────────────────────┐
│                                                          │
│   VaultNote 的 5 大核心竞争优势                          │
│                                                          │
│   1️⃣  唯一原生SwiftUI+内置加密+一次性买断               │
│       → 竞品要么不原生(Joplin/StandardNotes)，           │
│         要么不加密(Bear/Apple Notes)，                    │
│         要么订阅(Bear/Notability/Obsidian Sync)           │
│                                                          │
│   2️⃣  零配置加密（CryptoKit内置）                       │
│       → Obsidian需要安装第三方插件才能加密               │
│       → VaultNote开箱即加密，零学习成本                  │
│                                                          │
│   3️⃣  $4.99一次性 vs 竞品$36-$180/年                    │
│       → 比Bear便宜86%，比Evernote便宜97%                 │
│       → 与隐私用户"不愿月月付费"心理完美契合             │
│                                                          │
│   4️⃣  墨色模式（Ink Mode）独有                          │
│       → 专为长时间写作设计的暗色主题                     │
│       → 竞品无此功能，ASO差异化关键词                    │
│                                                          │
│   5️⃣  零依赖/零追踪/零账号                              │
│       → 100% Apple原生框架，无第三方SDK                  │
│       → 无Analytics，无Crashlytics                       │
│       → 隐私用户最看重的3个"零"                          │
│                                                          │
└──────────────────────────────────────────────────────────┘
```
